Niveau 3 — Filtrer, transformer, agréger

  • R
  • Python
library(dplyr)
resume <- barometre %>%
  filter(campus == "Paris") %>%
  mutate(satisfait = satisfaction >= 14) %>%
  group_by(annee_etude) %>%
  summarise(n = n(),
            taux_satis = mean(satisfait),
            satis_moy  = mean(satisfaction, na.rm = TRUE)) %>%
  arrange(desc(taux_satis))

Niveau 4 — Comparer deux vagues

  • R
  • Python
library(dplyr); library(tidyr)
vague_2024 <- read.csv("vague_2024.csv")
vague_2025 <- read.csv("vague_2025.csv")

evolution <- bind_rows(
    mutate(vague_2024, annee = 2024),
    mutate(vague_2025, annee = 2025)
  ) %>%
  group_by(annee, campus) %>%
  summarise(satis_moy = mean(satisfaction, na.rm = TRUE), .groups = "drop") %>%
  pivot_wider(names_from = annee, values_from = satis_moy) %>%
  mutate(evolution = `2025` - `2024`) %>%
  arrange(desc(evolution))

import pandas as pd
vague_2024 = pd.read_csv("vague_2024.csv"); vague_2024["annee"] = 2024
vague_2025 = pd.read_csv("vague_2025.csv"); vague_2025["annee"] = 2025
deux = pd.concat([vague_2024, vague_2025])

evolution = (deux.groupby(["annee", "campus"])["satisfaction"]
                 .mean()
                 .unstack("annee"))
evolution["evolution"] = evolution[2025] - evolution[2024]
evolution = evolution.sort_values("evolution", ascending=False)
print(evolution.round(2))
